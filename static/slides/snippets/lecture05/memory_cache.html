gameLoop(delta, absolute) {

  for(var entity in this.entities) {

    // 50% mispredictions

    if(entity.ACTIVE) { 

      entity.update(delta, absolute);

    }

    if(entity.VISIBLE) {

      entity.draw();

    }

  }

}
